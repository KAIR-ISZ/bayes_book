\tikzmath{
  function x(\x, \y, \z, \theta, \phi) {
    \xa = +cos(\theta) * \x + sin(\theta) * \z;
    \ya = \y;
    \za = -sin(\theta) * \x + cos(\theta) * \z;
    %
    \xb = \xa;
    \yb = +cos(\phi) * \ya + sin(\phi) * \za;
    %\zb = -sin(\phi) * \ya + cos(\phi) * \za;
    %  
    return \xb;
  };
  function y(\x, \y, \z, \theta, \phi) {
    \xa = +cos(\theta) * \x + sin(\theta) * \z;
    \ya = \y;
    \za = -sin(\theta) * \x + cos(\theta) * \z;
    %
    \xb = \xa;
    \yb = +cos(\phi) * \ya + sin(\phi) * \za;
    %\zb = -sin(\phi) * \ya + cos(\phi) * \za;
    %   
    return \yb;
  };
}
